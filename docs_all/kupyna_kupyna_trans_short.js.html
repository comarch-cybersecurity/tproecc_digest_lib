    G2(x, y, round) {
        this.G(x, y);
        for (var index = 0; index < 8; index++) {
            var addHi = 0x00f0f0f0 ^ ((((7 - index) * 16) ^ round) << 24);
            var result = this.addLongLong(y.hi[index], y.lo[index], addHi, 0xf0f0f0f3);
            y.hi[index] = result.hi;
            y.lo[index] = result.lo;   
        }
    }
